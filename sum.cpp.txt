#include <iostream>
using namespace std;
int main() {
    int n;
    cout << "Enter a 4-digit number: ";
    cin >> n;
    int d1 = n / 1000;
    int d2 = (n % 1000) / 100;
    int d3 = (n % 100) / 10;
    int d4 = n % 10;
    cout << "Sum of the individual digits: " << d1 + d2 + d3 + d4 << endl;
    return 0;
}
