#include <iostream>
using namespace std;
int main() {
    float u, a, t;
    cout << "Enter Initial Velocity (m/s): ";
    cin >> u;
    cout << "Enter Acceleration (m/s^2): ";
    cin >> a;
    cout << "Enter Time (s): ";
    cin >> t;
    cout << "Final Velocity (m/s): " << u + a * t << endl;
    return 0;
}
